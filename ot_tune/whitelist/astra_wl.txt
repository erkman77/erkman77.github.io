#!/bin/sh
_url=$(sed -n '1p' /tmp/read)
_urlsub=$(sed -n '2p' /tmp/read)

## define funtions

insert_enima2_settings () {

cp -pr /etc/enigma2/bouquets.tv /tmp/
sleep 3

exec 9>> /tmp/bouquets.tv
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.hdplus_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.dazn_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skybundesliga_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skydoku_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skyerotic_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skyfilm_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skykinder_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skymusik_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skyserien_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skysport_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skysportaustria_astra.tv" ORDER BY bouquet' >&9
echo '#SERVICE 1:7:1:0:0:0:0:0:0:0:FROM BOUQUET "userbouquet.skysportextra_astra.tv" ORDER BY bouquet' >&9

cp -pr /tmp/bouquets.tv /etc/enigma2/bouquets.tv

sleep 3

wget --quiet -O /etc/enigma2/userbouquet.hdplus_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.hdplus_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.dazn_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.dazn_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skybundesliga_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skybundesliga_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skydoku_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skydoku_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skyerotic_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skyerotic_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skyfilm_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skyfilm_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skykinder_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skykinder_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skymusik_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skymusik_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skyserien_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skyserien_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skysport_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skysport_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skysportaustria_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skysportaustria_astra.tv
wget --quiet -O /etc/enigma2/userbouquet.skysportextra_astra.tv ${_url}/${_urlsub}/ot_tune/astra/userbouquet.skysportextra_astra.tv

  rm /tmp/bouquets.tv
  echo "... done" 
}

## run functions

insert_enima2_settings
